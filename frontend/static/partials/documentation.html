Ball Maze Doc

Below are all of the functions that make up the ball maze game and what they do. 
this.update = function(time, maxspeed, acceleration, deceleration, bounciness) {
	....
}
This function is called every 10 milliseconds. It contains the meat of the physics of the game. The functions below are the one's called within the update function. 
function maxSpeed(velx, vely, maxspeed) {
	if (speedUp == true) {
		maxspeed += .5;
	}
	if (velx > maxspeed) {
		velx = maxspeed;
	}
	if (velx < -maxspeed) {
		velx = -maxspeed;
	}
	if (vely > maxspeed) {
		vely = maxspeed;
	}
	if (vely < -maxspeed) {
		vely = -maxspeed;
	}
	return [velx, vely];
}
This function is called to limit the balls max velocity. If the current velocity of the ball on the x or y axis surpasses the max velocity, it is made equal to the max velocity. The "speedUp" variable allows for a powerup to increase max velocity. 
function decelerate(speedx, speedy, decelRate) {
	if (speedx != 0 || speedy != 0) {
		if (speedx != 0) {
			speedx *= decelRate;
		}
		if (speedy != 0) {
			speedy *= decelRate;
		}
		return [speedx, speedy];
	}
	return false;
}
This function is called to steadily drop the ball's speed. If the ball's speed is anything but zero, it's speed is dropped by "decelRate". 
function wallBounce(x, y, speedx, speedy, r, bounciness) {
	if (x + r > (document.getElementById('svg').clientWidth)) {
		x = (document.getElementById('svg').clientWidth) - r;
		speedx *= -bounciness;
	}
	if (x - r < 0) {
		x = r;
		speedx *= -bounciness;
	}
	if (y + r > (document.getElementById('svg').clientHeight)) {
		y = (document.getElementById('svg').clientHeight) - r;
		speedy *= -bounciness;
	}
	if (y - r < 0) {
		y = r;
		speedy *= -bounciness;
	}
	return [x, y, speedx, speedy];
}
This function is called to check if the ball has hit an outer border. If the ball's edge is equal to or past the border, it's the location is set equal to the border and its speed is reversed and lessened by the amount of "bounciness" the border has. 
function keyPresses(speedx, speedy, acceleration){
	if (keyPressed[38] == true && keyPressed[39] == true) {
		speedy -= acceleration;
		speedx += acceleration;
	} else if (keyPressed[38] == true && keyPressed[37] == true) {
		speedy -= acceleration;
		speedx -= acceleration;
	} else if (keyPressed[40] == true && keyPressed[39] == true) {
		speedy += acceleration;
		speedx += acceleration;
	} else if (keyPressed[40] == true && keyPressed[37] == true) {
		speedy += acceleration;
		speedx -= acceleration;
	} else if (keyPressed[39] == true && keyPressed[37] != true) {
		speedx += acceleration;
	} else if (keyPressed[37] == true && keyPressed[39] != true) {
	 	speedx -= acceleration;

		speedy -= acceleration;

This function is called to grab the key presses for movemont. If the key for any direction is true, then acceleration is added to the ball's velocity. The if checks with two key presses are used for horizontal movement. 
function sizeChange(change) {
	var userBall = document.getElementById(html_id);
	if (change == 0){
		userBall.setAttribute('r', 20);
	} else {
		userBall.setAttribute('r', 10);
	}
}
This function checks if the inputted variable is active. If it is, it makes the ball larger. The variable is activated when the user passes over a blue square. 
function viewboxTransition(x, y, x2, y2) {
	var tranX = x - x2;
	var tranY = y - y2;
	tranX /= 300;
	tranY /= 300;
	var newX = x;
	var newY = y;
	var view = document.getElementById("svg");
	var i = 0;
	var tranInterval = setInterval(function() {
		if (i > 299) {
			clearInterval(tranInterval);
		}
		newX -= tranX;
		newY -= tranY;
		i++;
		view.setAttribute('viewBox', newX + " " + newY + " 300 300");
	}, 10);
	
}
This function is called to move the view box when the users ball passes through a teleport.
